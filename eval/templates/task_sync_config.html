Configuration Format

{
  "project_code": "Code-Data-Repack-SFT",
  "sheet_url": "https://docs.google.com/spreadsheets/d/...",
  "sync_interval_minutes": 60,
  "primary_key_column": "question_id",
  "scraping_needed": false,
  "link_column": "problem_link",
  "column_mapping": {
    "prompt": "Task Prompt",
    "reference_links": "Links",
    "status": "Status",
    "difficulty": "Difficulty Level"
  },
  "field_types": {
    "problem_link": "url",
    "reference_links": "links",
    "status": "badge",
    "difficulty": "text"
  },
  "display_config": {
    "field_order": ["question_id", "problem_link", "reference_links", "status", "difficulty"],
    "field_labels": {
      "question_id": "Question ID",
      "problem_link": "Problem Link",
      "reference_links": "Reference Links",
      "status": "Status",
      "difficulty": "Difficulty"
    }
  },
  "sync_mode": "prompt_in_sheet",
  "sheet_tab": "Sheet1",
  "is_active": true
}
project_code: Project identifier
sheet_url: Google Sheet URL
sync_interval_minutes: Sync frequency
primary_key_column: Unique identifier
scraping_needed: Enable scraping
link_column: Scraping source column
column_mapping: Field mappings
sync_mode: Synchronization mode
sheet_tab: Sheet tab name
is_active: Configuration status
